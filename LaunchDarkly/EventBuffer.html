def add_event(event)
  if @events.length < @capacity
    @events.push(event)
    @capacity_exceeded = false
  else
    @dropped_events += 1
    unless @capacity_exceeded
      @capacity_exceeded = true
      @logger.warn { "[LDClient] Exceeded event queue capacity. Increase capacity to avoid dropping events." }
    end
  end
end
