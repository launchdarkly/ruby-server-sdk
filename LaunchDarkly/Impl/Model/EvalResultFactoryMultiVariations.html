def for_variation(index, in_experiment)
  if index < 0 || index >= @factories.length
    EvaluationDetail.new(nil, nil, EvaluationReason.error(EvaluationReason::ERROR_MALFORMED_FLAG))
  else
    @factories[index].get_result(in_experiment)
  end
end
