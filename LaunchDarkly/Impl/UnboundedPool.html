def initialize(instance_creator, instance_destructor)

def acquire
  @lock.synchronize {
    if @pool.length == 0
      @instance_creator.call()
    else
      @pool.pop()
    end
  }
end

def dispose_all
  @lock.synchronize {
    @pool.map { |instance| @instance_destructor.call(instance) } unless @instance_destructor.nil?
    @pool.clear()
  }
end
